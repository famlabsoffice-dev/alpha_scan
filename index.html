function renderMatches(matches) { const display = document.getElementById('display'); if (!matches || matches.length === 0) { display.innerHTML = `
No Direct Matches Found
High market efficiency detected.
`; return; } let html = `
Arbitrage Opportunities: ${matches.length}
`; matches.forEach(match => { // Sicherstellen, dass platforms existiert, sonst Fallback const platforms = match.platforms ? match.platforms.join(' ↔ ') : 'Data Bridge'; html += `

${match.event}

MATCH
Bridge ${platforms}
Status ${match.potential}
Details ${match.details}
`; }); display.innerHTML = html; }
